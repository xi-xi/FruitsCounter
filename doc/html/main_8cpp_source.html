96 std::size_t getIncrementalTomato(const std::vector<cv::Rect>& previous_tomato, const std::vector<cv::Rect>& current_tomato, const COUNTUP_FUNC& coutup_func) {
97  if (previous_tomato.empty() || current_tomato.empty()) {
98  return 0;
99  }
100  std::size_t count = 0;
101  for (const auto& cur : current_tomato) {
102  double min_dist = std::numeric_limits<double>::infinity();
103  std::size_t min_dist_index = 0;
104  for (std::size_t i = 0; i < previous_tomato.size(); ++i) {
105  const auto& pre = previous_tomato[i];
106  const auto& pre_pos = ::rect2point(pre);
107  const auto& cur_pos = ::rect2point(cur);
108  double distance = std::sqrt(std::pow(pre_pos.x - cur_pos.x, 2) + std::pow(pre_pos.y - cur_pos.y, 2));
109  if (min_dist > distance) {
110  min_dist = distance;
111  min_dist_index = i;
112  }
113  }
114  if (!std::isinf(min_dist)){
115  if (coutup_func(::rect2point(cur), ::rect2point(previous_tomato[min_dist_index]))) {
116  count++;
117  }
118  }
119  }

121 }
